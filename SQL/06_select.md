# selecionar todas as colunas
~~~sql
SELECT [CODIGO DO PRODUTO]
      ,[NOME DO PRODUTO]
      ,[EMBALAGEM]
      ,[TAMANHO]
      ,[SABOR]
      ,[PRECO DE LISTA]
  FROM [TABELA DE PRODUTOS]

SELECT * FROM [TABELA DE CLIENTES]
~~~

# alias 
- ordem dos campos, será ordem de visualização
~~~sql
SELECT [CPF] AS ID,
[NOME] AS [NOME DO CLIENTE]
FROM [TABELA DE CLIENTES]

SELECT [MATRICULA] AS IDENTIFICADOR,
[NOME] AS [NOME DO VENDENDEDOR]
FROM [TABELA DE VENDEDORES]
~~~

# where (filtros)

~~~sql
SELECT * FROM [TABELA DE PRODUTOS] WHERE [CODIGO DO PRODUTO] ='1000889' 

SELECT [CODIGO DO PRODUTO], [NOME DO PRODUTO] FROM [TABELA DE PRODUTOS] WHERE [CODIGO DO PRODUTO] = '1002334'

SELECT * FROM [TABELA DE PRODUTOS] WHERE [SABOR] = 'Limão'

UPDATE [TABELA DE PRODUTOS] SET [PRECO DE LISTA] = 1.1 * [PRECO DE LISTA] WHERE [SABOR] = 'Limão'


UPDATE [TABELA DE VENDEDORES] SET [PERCENTUAL COMISSÃO] = 1.1*+[PERCENTUAL COMISSÃO] WHERE MATRICULA = '00239'
~~~

# Maior /menor que/ igual/ diferente

~~~sql
SELECT * FROM [TABELA DE PRODUTOS] WHERE EMBALAGEM = 'PET'

SELECT * FROM [TABELA DE PRODUTOS] WHERE [PRECO DE LISTA] <= 4.555

SELECT * FROM [TABELA DE PRODUTOS] WHERE [PRECO DE LISTA] < 10

SELECT * FROM [TABELA DE VENDEDORES] WHERE [PERCENTUAL COMISSÃO] > 0.10

SELECT * FROM [TABELA DE PRODUTOS] WHERE EMBALAGEM <= 'Lata'
--Neste caso, ele pega EMBALAGEM <=’Lata’, ele transforma isso internamente nos números da tabela ASCII e pega todos que tem valor na tabela ASCII menor do que ele.

SELECT * FROM [TABELA DE PRODUTOS] WHERE EMBALAGEM != 'Lata'

SELECT * FROM [TABELA DE CLIENTES] WHERE [DATA DE NASCIMENTO] = '1995-09-11'

SELECT * FROM [TABELA DE CLIENTES] WHERE [DATA DE NASCIMENTO] <= '1985-09-11'

SELECT * FROM [TABELA DE CLIENTES] WHERE YEAR([DATA DE NASCIMENTO]) >= 1995

SELECT * FROM [TABELA DE CLIENTES] WHERE MONTH([DATA DE NASCIMENTO])= 9
~~~
